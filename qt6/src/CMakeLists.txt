ADD_SUBDIRECTORY (icons)
ADD_SUBDIRECTORY (data)
ADD_SUBDIRECTORY (Antiqua)

SET (ANTIQUACRM_INCLUDE_DIRS
  ${ANTIQUACRM_INCLUDE_DIRS}
  ${CMAKE_CURRENT_SOURCE_DIR}
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}/Antiqua/Core
  ${CMAKE_CURRENT_SOURCE_DIR}/Antiqua/Widgets
  ${CMAKE_CURRENT_SOURCE_DIR}/Antiqua/Interface
)

INCLUDE_DIRECTORIES (${ANTIQUACRM_INCLUDE_DIRS})

QT6_ADD_RESOURCES (QRC_FILES resources.qrc)

SET (PROJECT_HEADERS
 test_widget.h
 application.h
 mainwindow.h
 menubar.h
 statusbar.h
)

SET (PROJECT_SOURCES
 test_widget.cpp
 main.cpp
 application.cpp
 mainwindow.cpp
 menubar.cpp
 statusbar.cpp
 ${QRC_FILES}
)

SET (PROJECT_LIBRARIES
  ${PROJECT_LIBRARIES}
  AntiquaCRM
  AntiquaWidgets
)

ADD_EXECUTABLE(${PROJECTNAME}
  ${PROJECT_HEADERS}
  ${PROJECT_SOURCES}
)

IF (Qt6WinExtras_FOUND)
  SET(PROJECT_LIBRARIES
   ${PROJECT_LIBRARIES}
   Qt6WinExtras
  )
ENDIF ()

TARGET_LINK_LIBRARIES(${PROJECTNAME} PRIVATE
  Qt6::Core
  Qt6::Gui
  Qt6::Network
  Qt6::Widgets
  Qt6::Sql
  Qt6::Xml
  ${PROJECT_LIBRARIES}
)

SET_TARGET_PROPERTIES(${PROJECTNAME} PROPERTIES
  OUTPUT_NAME ${PROJECTNAME}
  LINKER_LANGUAGE CXX
  WIN32_EXECUTABLE TRUE
)

INSTALL (TARGETS ${PROJECTNAME}
  RUNTIME DESTINATION ${CMAKE_INSTALL_PREFIX}/bin/
  COMPONENT Core
)

#EOF
